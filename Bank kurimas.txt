CREATE TABLE BANKAS(
BANK_ID SMALLINT PRIMARY KEY AUTO_INCREMENT NOT NULL,
PAVADINIMAS VARCHAR NOT NULL)

CREATE TABLE ASMUO(
ASMENS_KODAS BIGINT(11) PRIMARY KEY NOT NULL,
VARDAS VARCHAR NOT NULL,
PAVARDE VARCHAR NOT NULL);

CREATE TABLE SASKAITA(
SASKAITOS_NR VARCHAR PRIMARY KEY NOT NULL,
ASMENS_KODAS BIGINT(11) NOT NULL,
BANK_ID SMALLINT NOT NULL,
SASKAITOS_TIPAS VARCHAR DEFAULT 'DEBETINE',
FOREIGN KEY(ASMENS_KODAS) REFERENCES ASMUO(ASMENS_KODAS),
FOREIGN KEY(BANK_ID) REFERENCES BANKAS(BANK_ID));

CREATE TABLE PAVEDIMAI(
PAVEDIMO_NR BIGINT PRIMARY KEY NOT NULL,
SASKAITOS_NR_FROM VARCHAR NOT NULL,
SASKAITOS_NR_TO VARCHAR NOT NULL,
DATA DATE,
SUMA REAL,
FOREIGN KEY (SASKAITOS_NR_FROM) REFERENCES SASKAITA(SASKAITOS_NR),
FOREIGN KEY (SASKAITOS_NR_TO) REFERENCES SASKAITA(SASKAITOS_NR));

CREATE TABLE INESIMAS(
SASKAITOS_NR VARCHAR NOT NULL,
DATA DATE,
SUMA REAL,
FOREIGN KEY(SASKAITOS_NR) REFERENCES SASKAITA(SASKAITOS_NR));

CREATE TABLE ISEMIMAS(
SASKAITOS_NR VARCHAR NOT NULL,
DATA DATE,
SUMA REAL,
FOREIGN KEY(SASKAITOS_NR) REFERENCES SASKAITA(SASKAITOS_NR));
